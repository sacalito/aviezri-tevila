# Calculate seconds until next 10-minute mark
def seconds_until_next_interval(interval=10):
    now = datetime.datetime.now()
    minutes = now.minute
    seconds = now.second
    microseconds = now.microsecond
    
    # Calculate minutes until next interval
    minutes_to_add = interval - (minutes % interval)
    if minutes_to_add == interval:
        minutes_to_add = 0
    
    # Calculate total seconds to wait
    total_seconds = minutes_to_add * 60 - seconds - microseconds/1000000
    
    if total_seconds < 0:
        total_seconds = 0
    
    return total_seconds

# Timer function to run check_operating_hours at specific intervals
def scheduler_thread():
    logger.info("Scheduler thread started")
    
    while True:
        try:
            # Calculate wait time until next 10-minute mark
            wait_time = seconds_until_next_interval(10)
            
            # Ensure we always wait at least 1 second to avoid infinite loops
            if wait_time < 1:
                wait_time = 1  # Minimum wait of 1 second
                
            logger.info(f"Waiting for {wait_time:.2f} seconds until next check")
            
            # Sleep until next interval
            time.sleep(wait_time)
            
            # Run the check
            with app.app_context():
                check_operating_hours()
                
        except Exception as e:
            logger.error(f"Error in scheduler thread: {e}")
            # Wait a bit and try again
            time.sleep(60)

# Start the scheduler thread
scheduler = threading.Thread(target=scheduler_thread)
scheduler.daemon = True  # This makes the thread exit when the main program exits
